163 /*!
164  * @fn SnesCtrl snesCtrlSetup(int port, int dataPin, int clkPin, int latchPin)
165  *
166  * @brief Initializes the control structure and the peripherals used by the
167  * controller.
168  *
169  * This function initializes the control structure (SnesCtrl), port and pins
170  * used by the controller.
171  *
172  * __Example__
173  *
174  * // This example shows how to configure and operate 2 gamepad, being the
175  * // first one connected to port 1, pins 1, 2 and 3 (data, clock and latch
176  * // respectively), and the second one to port 2, pins 1, 2 and 3 (data,
177  * // clock and latch respectively).
178  *
179  * // Create the controller structures.
180  * SnesCtrl controller1, controller2;
181  *
182  * // Configure and initialize the first gamepad.
183  * controller1 = snesCtrlSetup(1, 1, 2, 3);
184  * // Configure and initialize the second gamepad.
185  * controller2 = snesCtrlSetup(2, 1, 2, 3);
186  *
187  * // Read each gamepad B button and prints a message.
188  * SnesCtrlBtnPressed buttons1, buttons2;
189  * buttons1 = snesCtrlRead(&controller1);
190  * buttons2 = snesCtrlRead(&controller2);
191  *
192  * if(buttons1.b || buttons2.b)
193  * printf("Button B was pressed\n");
194  *
195  * @param[in] port Port number (only port 1 or 2 is allowed).
196  * @param[in] dataPin Data line pin number (0 to 7).
197  * @param[in] clkPin Clock line pin number (0 to 7).
198  * @param[in] latchPin Latch line pin number (0 to 7).
199  *
200  * @return Returns an initialized SnesCtrl structure.
201  */
202 SnesCtrl snesCtrlSetup(int port, int dataPin, int clkPin, int latchPin);

223  */
224 SnesCtrlBtnPressed snesCtrlRead(SnesCtrl *ctrl);
