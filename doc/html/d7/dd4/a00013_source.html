41 // Internal function that triggers the latch.
42 void _snesCtrlLatchPulse(SnesCtrl *ctrl) {
43  // Trigger the latch for 12us.
44  *(ctrl->portOut) |= ctrl->latchPin;
45  DELAY_12us;
46  *(ctrl->portOut) &= ~ctrl->latchPin;
47 }

49 // Internal function that reads one button status from the controller.
50 bool _snesBitRead(SnesCtrl *ctrl) {
51  // Wait until half clock cycle.
52  DELAY_6us;
53 
54  bool status = false;
55 
56  // Read the data pin state.
57  // 1 mean the button is not pressed and 0 means the button is pressed.
58  status = ((*(ctrl->portIn) & ctrl->dataPin) > 0) ? false : true;
59 
60  // 6us clock pulse.
61  *(ctrl->portOut) &= ~ctrl->clockPin;
62  DELAY_6us;
63  *(ctrl->portOut) |= ctrl->clockPin;
64 
65  return status;
66 }

68 SnesCtrl snesCtrlSetup(int port, int dataPin, int clkPin, int latchPin) {
69  SnesCtrl ctrl;
70 
71  // Select port 1 or 2 only. If another port is used, this function needs to
72  // be extended.
73  switch (port) {
74  case 1:
75  ctrl.portDir = (unsigned char *) &(P1DIR);
76  ctrl.portOut = (unsigned char *) &(P1OUT);
77  ctrl.portIn = (unsigned char *) &(P1IN);
78  break;
79  case 2:
80  ctrl.portDir = (unsigned char *) &(P2DIR);
81  ctrl.portOut = (unsigned char *) &(P2OUT);
82  ctrl.portIn = (unsigned char *) &(P2IN);
83  break;
84  default:
85  ctrl.portDir = 0;
86  ctrl.portOut = 0;
87  ctrl.portIn = 0;
88  }
89 
90  // Map from integer to bit mask.
91  ctrl.dataPin = 1 << dataPin;
92  ctrl.clockPin = 1 << clkPin;
93  ctrl.latchPin = 1 << latchPin;
94 
95  // Setup clock and latch initial state.
96  // Clock is normally high.
97  *(ctrl.portOut) |= ctrl.clockPin;
98  // Latch is normally low.
99  *(ctrl.portOut) &= ~ctrl.latchPin;
100 
101  // Configure data, latch and clock pins as input, output and output,
102  // respectively.
103  *(ctrl.portDir) &= ~ctrl.dataPin;
104  *(ctrl.portDir) |= ctrl.latchPin + ctrl.clockPin;
105 
106  // Waits a little bit until thing set up.
107  DELAY_12us;
108 
109  return ctrl;
110 }
